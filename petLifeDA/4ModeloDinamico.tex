\section{Actores del sistema}
	

%---------------------------------------------------------
\begin{Usuario}{\hypertarget{JefePosgrado}{\subsection{Jefe de la Unidad de Posgrado}}}{
		Es el encargado de dirigir el área de posgrado, para fines del sistema es la máxima autoridad y quien resuelve cualquier controversia que se presente conforme a reglamento y o que disponga el Colegio de profesores.
	}
	\item[Responsabilidades:] Se consideran relevantes para el diseño del sistema las siguientes:
	\begin{itemize}
		\item Supervisar el proceso de inscripción y reinscripción.
		\item Resolver controversias.
		\item Autorizar inscripciones para casos especiales.
		\item dar seguimiento a alumnos extranjeros y en movilidad.
	\end{itemize}
	
	\item[Perfil:] \cdtEmpty
	\begin{itemize}
		\item Tener nombramiento para el puesto.
		\item Tener una licenciatura en el área de conocimiento.
		\item Conocimientos sobre los reglamentos de la universidad, del área de posgrado y de ciencia y tecnología.
		\item Conocimientos básicos de informática y administración.
	\end{itemize}
	\item[Procesos en los que participa:] \cdtEmpty
	\begin{itemize}
		\item PRS-01 Gestión de Programas Académicos.
		\item PRS-02 Gestión de Unidad de Aprendizaje.
		\item PRS-04 Gestión de personal de control escolar.
		\item PRC-02 Planeación del ciclo escolar.
	\end{itemize}
\end{Usuario}

%---------------------------------------------------------
\begin{Usuario}{\hypertarget{JefeControlEscolar}{\subsection{Jefe de control escolar}}}{
		Encargado de llevar todos movimientos de los alumnos respecto a su carga académica y registro de calificaciones con el apoyo del \hyperlink{ControlEscolar}{Personal de Control Escolar}.
	}
	\item[Responsabilidades:] Se consideran relevantes para el diseño del sistema las siguientes:
	\begin{itemize}
		\item Coordinar y supervisar los procesos de inscripción y reinscripción.
		\item Coordinar y supervisar el registro de alumnos extranjeros.
		\item Coordinar y supervisar el registrar equivalencias para alumnos en movilidad.
		\item Coordinar y supervisar todos los movimientos de carga académica.
		\item Realizar informes mensuales.
		\item Coordinar y supervisar la emisión de constancias a los alumnos relativas a su situación escolar.
		\item Coordinar y supervisar la emisión de credenciales.
	\end{itemize}
	
	\item[Perfil:] \cdtEmpty
	\begin{itemize}
		\item Contar con su nombramiento otorgado por el Director.
		\item Experiencia en el área de posgrado.
		\item Conocimientos sobre los reglamentos de la universidad, del área de posgrado y de ciencia y tecnología.
		\item Conocimientos básicos de informática y administración.
	\end{itemize}
	\item[Procesos en los que participa:] \cdtEmpty
	\begin{itemize}
		\item PRS-01 Gestión de Programas Académicos.
		\item PRS-02 Gestión de Unidad de Aprendizaje.
		\item PRS-03 Gestión de Docentes.
		\item PRS-04 Gestión de personal de control escolar.
		\item PRC-01 Planeación académica.
		\item PRC-02 Planeación del ciclo escolar.
		\item PRC-03 Inscripción de alumno de nuevo ingreso.
		\item PRC-04 Reinscripción de alumno.
		\item PRC-05 Movimiento de Unidad de Aprendizaje.
		\item PRC-06 Registro de calificaciones.
		\item PRC-07 Generación de credenciales.
		\item PRC-08 Generación de constancias.
		\item PRC-09 Gestión de pagos de servicios.
		\item PRC-10 Atención a alumnos de movilidad.
		\item PRC-11 Generación de reportes.
	\end{itemize}
\end{Usuario}
